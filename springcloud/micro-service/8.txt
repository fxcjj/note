Spring Cloud Zuul
	* 路由+过滤器=zuul
	* 一系列过滤器

过滤器
	pre filter
	routing filter
	post filter //结果的处理或加工
	error filter //上面三个发生异常时走过个
	custom filter //自定义filter

zuul:
	routes:
		aaa:
			path: /myProduct/**
			serviceId: product
			sensitiveHeaders: //有cookie
			#简洁写法
			#product: /myProduct/**
	#排除某些路由set
	ignored-patterns:
		- /**/product/listForOrder
management:
	security:
		enabled: false

		
ZuulConfig
	@ConfigurationProperties("zuul")
	@RefreshScope
	public ZuulProperties zuulProperties() {
		new ZuulProperties();
	}
